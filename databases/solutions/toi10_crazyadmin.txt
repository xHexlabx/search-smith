#include <bits/stdc++.h>
using namespace std;

const int N = 100;
int A[N];
int n,m,mx;

int main(){
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);

    cin >> n >> m;
    for(int i=0;i<m;i++)cin >> A[i],mx=max(mx,A[i]);

    int l = mx,r = 1e7,mid;
    while(l<r){
        mid = (l+r) >> 1;

        int cnt = 0,kps = 0;
        for(int i=0;i<m;i++){
            if(kps+A[i]<=mid)kps += A[i];
            else kps = A[i],cnt++;
        }

        if(cnt>=n)l = mid+1;
        else r = mid;
    }
    cout << l;

    return 0;
}
